# Vue.js Integration

Use Signal JS with Vue.js by installing the `@signal-js/browser` package directly.

## Installation

```bash
npm install @signal-js/browser
```

## Setup

Initialize Signal JS in your Vue application:

```vue
<script setup>
import { onMounted } from 'vue'
import { createSignal } from '@signal-js/browser'

let signal = null

onMounted(async () => {
  signal = createSignal({
    endpoint: 'https://your-api.com/api/signal/v1',
    apiKey: 'your-api-key',
    projectId: 'your-project-id',
    // Enable tracing headers for server correlation
    addTracingHeaders: true, // or ['api.example.com'] for specific hostnames
  })
  
  await signal.start()
})

// Identify users
function identifyUser(userId, properties) {
  signal?.identify(userId, properties)
}

// Track events
function trackEvent(eventName, properties) {
  signal?.capture(eventName, properties)
}
</script>

<template>
  <div>
    <button @click="trackEvent('button_clicked', { buttonId: 'cta' })">
      Click Me
    </button>
  </div>
</template>
```

## Vue Router Integration

Track page views with Vue Router:

```vue
<script setup>
import { onMounted, watch } from 'vue'
import { useRoute } from 'vue-router'
import { createSignal } from '@signal-js/browser'

const route = useRoute()
let signal = null

onMounted(async () => {
  signal = createSignal({
    endpoint: 'https://your-api.com/api/signal/v1',
    apiKey: 'your-api-key',
    projectId: 'your-project-id',
    capturePageview: false, // Disable auto pageview, track manually
  })
  
  await signal.start()
  
  // Track initial page view
  signal.capturePageview({
    path: route.path,
    title: document.title,
  })
})

// Track page views on route changes
watch(() => route.path, (newPath) => {
  signal?.capturePageview({
    path: newPath,
    title: document.title,
  })
})
</script>
```

## Composable Pattern

Create a composable for easier use:

```typescript
// composables/useSignal.ts
import { ref, onMounted } from 'vue'
import { createSignal } from '@signal-js/browser'

let signalInstance = null

export function useSignal() {
  const isReady = ref(false)
  
  onMounted(async () => {
    if (!signalInstance) {
      signalInstance = createSignal({
        endpoint: import.meta.env.VITE_SIGNAL_ENDPOINT,
        apiKey: import.meta.env.VITE_SIGNAL_API_KEY,
        projectId: import.meta.env.VITE_SIGNAL_PROJECT_ID,
      })
      
      await signalInstance.start()
      isReady.value = true
    }
  })
  
  return {
    signal: signalInstance,
    isReady,
    identify: (userId, properties) => signalInstance?.identify(userId, properties),
    capture: (eventName, properties) => signalInstance?.capture(eventName, properties),
    capturePageview: (options) => signalInstance?.capturePageview(options),
  }
}
```

Usage in components:

```vue
<script setup>
import { useSignal } from '@/composables/useSignal'

const { identify, capture } = useSignal()

function handleLogin(user) {
  identify(user.id, { email: user.email })
}

function handleButtonClick() {
  capture('button_clicked', { buttonId: 'cta' })
}
</script>
```

## Complete Example

```vue
<script setup>
import { onMounted } from 'vue'
import { useRouter } from 'vue-router'
import { createSignal } from '@signal-js/browser'

const router = useRouter()
let signal = null

onMounted(async () => {
  signal = createSignal({
    endpoint: import.meta.env.VITE_SIGNAL_ENDPOINT,
    apiKey: import.meta.env.VITE_SIGNAL_API_KEY,
    projectId: import.meta.env.VITE_SIGNAL_PROJECT_ID,
    addTracingHeaders: true,
    capturePageview: true,
    capturePageleave: true,
  })
  
  await signal.start()
})

function trackPurchase(productId, amount) {
  signal?.capture('purchase_completed', {
    productId,
    amount,
    currency: 'USD',
  })
}
</script>

<template>
  <div>
    <button @click="trackPurchase('prod_123', 99.99)">
      Buy Now
    </button>
  </div>
</template>
```

## Best Practices

1. **Initialize once** - Create the Signal instance once in your app entry point
2. **Use composables** - Create a composable for reusable Signal functionality
3. **Track route changes** - Use Vue Router's navigation guards or watchers to track page views
4. **Enable tracing headers** - Set `addTracingHeaders: true` if you have server-side tracking
5. **Use environment variables** - Store credentials in `.env` files
